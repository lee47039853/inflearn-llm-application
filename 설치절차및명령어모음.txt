0. pyenv 설치 절차

# GIT을 통해 pyenv 다운로드
git clone https://github.com/pyenv-win/pyenv-win.git .

# 시스템 환경변수 생성 및 path 추가
PYENV_ROOT = C:\PYENV\pyenv-win
%PYENV_ROOT%\bin
%PYENV_ROOT%\libexec

# 설치 가능한 Python 버전 목록 확인, 필터링, 특정 버전 설치
pyenv install --list
pyenv install --list | findstr "3.11"
pyenv install 3.11.6

# 설치 확인
pyenv versions
pyenv exec python --version

1. 현재 디넥토리 어플리테이션으로 하는 가상환경 구성
pyenv exec python -m venv .

2. 가상환경 활성화 및 비활성화
.\Scripts\Activate.ps1
